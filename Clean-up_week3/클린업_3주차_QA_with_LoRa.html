PEFT란?¶

PEFT란 Parameter-Efficient Fine Tuning의 약자로, 사전 훈련된 LLM (Large Language Model)을

특정 상황에 적용할 때, 대부분의 파라미터를 freeze하고, 소수의 모델 파라미터만 fine-tuning하는 기법입니다.

모델이 점점 거대해지면서 일반 GPU로는 파인튜닝이 어렵기 때문에 나온 개념이라고 볼 수 있습니다.

PEFT의 장점은 다음과 같습니다.

  1. 매우 적은 학습 비용 - 우리는 PEFT를 통해 당연히 매우 적은 비용으로도 학습이 가능해집니다.

  2. Catastrophic forgetting 문제 방지 - 이 문제는 모델이 이전에 학습했던 데이터들을 잊어버리는 현상입니다.

    그러나 PEFT는 매우 적은 파라미터만 파인튜닝하기 때문에 이를 방지할 수 있습니다.

다양한 PEFT 방법들이 존재하지만 대표적인 기법은 LoRa입니다.

LoRa는 Low-Rank Adaptation의 약자로 두 개의 저차원의 행렬로 근사한 후 파인튜닝하는 방법입니다.

def print_json_tree(data, indent=""):
    for key, value in data.items():
        if type(value) == list:
            print(f'{indent}- {key}: [{len(value)}]')
            print_json_tree(value[0], indent + "  ")
        else:
            print(f'{indent}- {key}: {value}')

파일은 Question-Context-Answer 구조로 되어있습니다.

한 번 요소들을 정리해보겠습니다.

  • question: context에 대한 질문입니다.

  • context: 질문에 해당하는 답을 찾을 수 있는 본문입니다.

  • answer: question에 대한 답입니다.

  • answer_start: 본문에서 answer가 시작하는 위치입니다.

보다시피 질문을 받으면 context에서 답을 찾고 있습니다.

def read_koquad(path):
  with open(path) as f:
    data = json.load(f)

  contexts = []
  questions = []
  answers = []

  for group in data["data"]:
    for passage in group["paragraphs"]:
      context = passage["context"]
      for qa in passage["qas"]:
        question = qa["question"]
        for answer in qa["answers"]:
          answers.append(answer)
          contexts.append(context)
          questions.append(question)
  return contexts, questions, answers

train_contexts, train_questions, train_answers = read_koquad("train/KorQuAD_v1.0_train.json")
dev_contexts, dev_questions, dev_answers = read_koquad("dev/KorQuAD_v1.0_dev.json")

def add_end_idx(answers):
  """
  answers에 answer_end를 추가해주는 함수
  """
  for answer in answers:
    answer["text"] = answer["text"].rstrip()  # 오른쪽 공백 제거
    start_idx = answer["answer_start"]
    end_idx = start_idx + len(answer["text"])
    answer["answer_end"] = end_idx

def add_token_positions(encodings, answers) -> list[int]:
    start_positions = []
    end_positions = []
    deleting_list = []

    for i in tqdm(range(len(answers))):
        # 시작 인덱스, 종료 인덱스 찾기
        start_position = encodings.char_to_token(i, answers[i]["answer_start"])
        end_position = encodings.char_to_token(i, answers[i]["answer_end"] - 1)

        start_positions.append(start_position)
        end_positions.append(end_position)

        # 시작 인덱스가 비정상, 즉 본문에 정답이 없는 경우
        if start_positions[-1] is None:
            start_positions[-1] = tokenizer.model_max_length - 1
            deleting_list.append(i)

        # 종료 인덱스가 비정상, 즉 본문에 정답이 없는 경우
        if end_positions[-1] is None:
            end_positions[-1] = tokenizer.model_max_length - 1
            deleting_list.append(i)

    encodings["start_positions"] = start_positions
    encodings["end_positions"] = end_positions
    return list(set(deleting_list))   # 중복 방지

import torch.nn as nn

class BertModelQA(nn.Module):
    """
    QA를 위한 BertModel입니다.
    start_index, end_index를 예측해야 하므로 2개의 뉴런을 갖는
    qa_outputs 층을 만들어주었습니다.
    """
    def __init__(self, model_name):
        super(BertModelQA, self).__init__()
        self.bert = AutoModel.from_pretrained(model_name)
        self.config = self.bert.config
        self.qa_outputs = nn.Linear(self.bert.config.hidden_size, 2)

    def forward(self, input_ids, attention_mask=None, inputs_embeds=None):
        # outputs[0].shape = (batch_size, sentence_length, 768)
        # 여기서는 (32, 512, 768)
        outputs = self.bert(input_ids, attention_mask)
        sequence_output = outputs[0]

        # (start, end) logit 값을 담은 Output 생성
        logits = self.qa_outputs(sequence_output)  # (32, 512, 2)
        start_logits, end_logits = logits.split(1, dim=-1)  # (32, 512, 1)

        start_logits = start_logits.squeeze(-1)  # (32, 512)
        end_logits = end_logits.squeeze(-1)

        # softmax를 거치면서 확률로 변환
        start_probs = torch.softmax(start_logits, dim=-1)
        end_probs = torch.softmax(end_logits, dim=-1)

        return start_probs, end_probs

def predict_and_evaluate(model, dataloader, loss_fn, is_return=False):
    """
    dataloader를 돌면서 예측 인덱스 값을 얻는 predict와
    평균 손실 값 및 정확도를 계산하는 evaluate를 하는 함수
    is_return=True일 경우, 예측 인덱스를 반환함
    """
    length = len(dataloader) * dataloader.batch_size
    total_loss = 0
    total_acc = 0
    pred_start_index = torch.tensor([], dtype=torch.int).to(device)
    pred_end_index = torch.tensor([], dtype=torch.int).to(device)

    for input_ids, attention_masks, start_positions, end_positions in tqdm(dataloader):
        input_ids = input_ids.to(device)
        attention_masks = attention_masks.to(device)
        start_positions = start_positions.to(device)
        end_positions = end_positions.to(device)

        # 기울기 변화 X
        with torch.no_grad():
          # 모델 예측
          start_probs, end_probs = model(input_ids, attention_masks)

          # 예측 인덱스 반환
          start_index = torch.argmax(start_probs, dim=-1)
          end_index = torch.argmax(end_probs, dim=-1)

          pred_start_index = torch.cat((pred_start_index, start_index), dim=0)
          pred_end_index = torch.cat((pred_end_index, end_index), dim=0)

          # 둘이 동시에 같을 때에만 맞는 것
          correct_start = start_index == start_positions
          correct_end = end_index == end_positions
          correct = torch.sum(correct_start * correct_end)
          total_acc += correct.item()

          # 손실 값 계산
          start_loss = loss_fn(start_probs, start_positions)
          end_loss = loss_fn(end_probs, end_positions)

          # Loss의 평균을 더함
          loss = (start_loss + end_loss) / 2
          total_loss += loss.item()
    avg_loss = total_loss / length
    avg_acc = total_acc / length
    print()
    print(f"Avg Test Loss - {avg_loss} / Avg Test Acc - {avg_acc * 100:.2f}%")

    if is_return:
      outputs = (pred_start_index, pred_end_index)
    else:
      outputs = 0
    return outputs

    """
    Prints the number of trainable parameters in the model.
    """
    trainable_params = 0
    all_param = 0
    for _, param in model.named_parameters():
        all_param += param.numel()
        if param.requires_grad:
            trainable_params += param.numel()
    print(
        f"trainable params: {trainable_params} || all params: {all_param} || trainable%: {100 * trainable_params / all_param}"
